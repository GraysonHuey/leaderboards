<div id="leaderboard-container"> <iframe id="leaderboard-iframe" src="https://graysonhuey.github.io/leaderboards/" width="100%" height="600" frameborder="0" title="WARU Band Olympics Leaderboard" allow="clipboard-read; clipboard-write" style="border: none; display: block;"> </iframe> </div> <script> // Auto-resize iframe based on content window.addEventListener('message', function(event) { // Verify origin for security if (event.origin !== 'https://graysonhuey.github.io') return; const iframe = document.getElementById('leaderboard-iframe'); if (!iframe) return; // Handle different message types switch (event.data.type) { case 'IFRAME_READY': case 'HEIGHT_CHANGE': case 'RESIZE_RESPONSE': if (event.data.height) { iframe.style.height = event.data.height + 'px'; } break; } }); // Request initial size document.getElementById('leaderboard-iframe').onload = function() { this.contentWindow.postMessage({ type: 'RESIZE_REQUEST' }, 'https://graysonhuey.github.io'); }; </script>
